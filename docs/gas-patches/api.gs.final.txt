 *     stampEndAt     … スタンプ取得終了日時 (例: 2026/08/23 17:00)  ← T3追加
 *     exchangeDeadline … 景品交換期限       (例: 2026/08/23 17:30)  ← T3追加
 */

function doGet(e) {
  const result = dispatch_(e);
  return ContentService
    .createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

// POST エントリ（事前登録・ファイルアップロード等。
// CORSプリフライト回避のためフロントは Content-Type を付けず JSON文字列を body で送る）
function doPost(e) {
  let p = (e && e.parameter) ? e.parameter : {};
  try {
    if (e && e.postData && e.postData.contents) {
      const j = JSON.parse(e.postData.contents);
      p = Object.assign({}, p, j);
    }
  } catch (err) { /* JSONでなければ parameter のみ使用 */ }
  const result = dispatch_({ parameter: p });
  return ContentService
    .createTextOutput(JSON.stringify(result))
    .setMimeType(ContentService.MimeType.JSON);
}

// ============================================================
// アクション: getStudent (企業向け / cardToken)
// ============================================================

function actionGetStudent_(p) {
  if (!p.token) return err_('missing_token');
  if (!p.event) return err_('missing_event');

  // 学生自身のMYPASS表示（source=mypass）はイベント期間中のみ許可する。
  // 企業が学生カードを確認する場合はsourceなし → publicDeadlineまで許可。
  if (p.source === 'mypass' && !isEventDay_(p.event)) return err_('event_ended');

  const cfg = getConfig_(p.event);
  if (isPastDeadline_(cfg)) return err_('expired');

  const student = findStudentByCardToken_(p.token, p.event);
  if (!student) return err_('invalid_token');

  // MYPASS からのリクエストはスタンプトークンも返す（進捗リンクに ?st= を付与するため）
  let stampToken = null;
  if (p.source === 'mypass') {
    const ps = eventSs_(p.event).getSheetByName(SHEET.STAMP_PARTICIPANTS);
    if (ps && ps.getLastRow() > 1) {
      const pd  = ps.getDataRange().getValues();
      const ph  = pd[0].map(String);
      const pEv = ph.indexOf('eventId');
      const pSi = ph.indexOf('studentId');
      const pSt = ph.indexOf('stampToken');
      const row = pd.slice(1).find(r =>
        String(r[pEv]) === String(p.event) &&
        String(r[pSi]) === String(student.studentId)
      );
      if (row && pSt >= 0) stampToken = String(row[pSt]);
    }
  }

  return ok_({
    studentId:  student.studentId,
    name:       student.name,
    furigana:   student.furigana,
    school:     student.school,
    department: student.department,
    year:       student.year,
    category:   student.category,
    clubYears:  student.clubYears,
    prefecture: student.prefecture,
    birthday:   student.birthday,
    email:      student.email,
    eventName:  cfg.eventName,
    deadline:   cfg.publicDeadline,
    stampToken: stampToken,
  });
}

// ============================================================
// アクション: saveViewLog (企業向け)
// 企業識別は2系統:
//   vk(viewKey)あり → サーバ側で企業IDに解決して自動記録 (source: company_auto)
//   companyあり     → 従来のメール手動登録等          (source: manual_email)
// VIEW_LOG 6列目に source を記録して両者を区別する。
// ============================================================

function actionSaveViewLog_(p) {
  if (!p.token) return err_('missing_token');
  if (!p.event) return err_('missing_event');

  const cfg = getConfig_(p.event);
  if (isPastDeadline_(cfg)) return err_('expired');

  const student = findStudentByCardToken_(p.token, p.event);
  if (!student) return err_('invalid_token');

  let companyId = p.company || '';
  let source    = 'manual_email';
  if (p.vk) {
    const company = findCompanyByViewKey_(p.vk, p.event);
    if (!company) return err_('invalid_key');
    companyId = company.companyId;
    source    = 'company_auto';
  }
  if (!companyId) return err_('missing_params');

  const sheet = eventSs_(p.event).getSheetByName(SHEET.VIEW_LOG);

  // 自動記録の軽い重複抑制:
  // 同一イベント×同一学生×同一企業の行が既にあれば時刻のみ更新(行は増やさない)
  if (source === 'company_auto' && sheet.getLastRow() > 1) {
    const rows = sheet.getRange(2, 1, sheet.getLastRow() - 1, 4).getValues();
    const idx = rows.findIndex(r =>
      String(r[1]) === String(p.event) &&
      String(r[2]) === String(student.studentId) &&
      String(r[3]) === String(companyId)
    );
    if (idx >= 0) {
      sheet.getRange(idx + 2, 1).setValue(new Date());
      return ok_({ logged: true, updated: true });
    }
  }

  sheet.appendRow([
    new Date(), p.event, student.studentId, companyId, p.email || '', source,
  ]);

  return ok_({ logged: true });
}

// ============================================================
// アクション: activateStamp (cardToken → stampToken)
// ============================================================

function actionActivateStamp_(p) {
  if (!p.token) return err_('missing_token');
  if (!p.event) return err_('missing_event');
  if (isEventInactive_(p.event)) return err_('event_inactive');

  const cfg = getConfig_(p.event);
  if (!isStampOpen_(cfg)) return err_('stamp_closed');

  const student = findStudentByCardToken_(p.token, p.event);
  if (!student) return err_('invalid_token');

  const lock = LockService.getScriptLock();
  try { lock.waitLock(6000); } catch (e) { return err_('lock_timeout'); }
  try {
    const ss_ = eventSs_(p.event);
    let participantSheet = ss_.getSheetByName(SHEET.STAMP_PARTICIPANTS);
    if (!participantSheet) {
      participantSheet = ss_.insertSheet(SHEET.STAMP_PARTICIPANTS);
      participantSheet.appendRow(['時刻', 'eventId', 'studentId', 'stampToken', '開始日時', 'status']);
    }

    let stampToken = null;
    if (participantSheet.getLastRow() > 1) {
      const data    = participantSheet.getDataRange().getValues();
      const headers = data[0].map(String);
      const evCol   = headers.indexOf('eventId');
      const sidCol  = headers.indexOf('studentId');
      const stCol   = headers.indexOf('stampToken');
      const existing = data.slice(1).find(r =>
        String(r[evCol]) === String(p.event) &&
        String(r[sidCol]) === String(student.studentId)
      );
      if (existing) stampToken = String(existing[stCol]);
    }

    let isNew = false;
    if (!stampToken) {
      stampToken = Utilities.getUuid();
      isNew = true;
      participantSheet.appendRow([
        new Date(), p.event, student.studentId,
        stampToken, new Date(), 'active',
      ]);
    }

    const count = countStamps_(p.event, student.studentId);
    return ok_(buildStampResult_(p.event, student.studentId, count, { stampToken, isNew }));
  } finally {
    lock.releaseLock();
  }
}

// ============================================================
// アクション: saveStamp (stampToken)
// ============================================================

function actionSaveStamp_(p) {
  if (!p.st || !p.ct) return err_('missing_params');
  if (!p.event) return err_('missing_event');
  if (isEventInactive_(p.event)) return err_('event_inactive');
  const cfg = getConfig_(p.event);
  if (!isStampOpen_(cfg)) return err_('stamp_closed');

  const student = findStudentByStampToken_(p.st, p.event);
  if (!student) return err_('invalid_student_token');

  const company = findCompanyByStampKey_(p.ct, p.event);
  if (!company) return err_('invalid_company_key');

  const lock = LockService.getScriptLock();
  try { lock.waitLock(6000); } catch (e) { return err_('lock_timeout'); }
  try {
    const sheet = eventSs_(p.event).getSheetByName(SHEET.STAMP_LOG);
    const rows  = sheet.getLastRow() > 1
      ? sheet.getRange(2, 1, sheet.getLastRow() - 1, 9).getValues()
      : [];

    const dup = rows.some(r =>
      String(r[1]) === String(p.event) &&
      String(r[2]) === String(student.studentId) &&
      String(r[7]) === String(company.companyId)
    );
    if (dup) return err_('already_stamped');

    if (p.nc) {
      const ncUsed = rows.some(r =>
        String(r[7]) === String(company.companyId) &&
        String(r[8]) === String(p.nc)
      );
      if (ncUsed) return err_('used_nfc');
    }

    sheet.appendRow([
      new Date(), p.event, student.studentId,
      student.name, student.school, student.category,
      company.companyName, company.companyId, p.nc || '',
    ]);

    const prevCount = rows.filter(r =>
      String(r[1]) === String(p.event) && String(r[2]) === String(student.studentId)
    ).length;
    const count = prevCount + 1;

    // 今回のスタンプで景品条件を達成したか (cfg は上で取得済み)
    const threshold = Number(cfg.prizeThreshold) || Number(cfg.prizeCriteria) || 5;
    const justCleared = prevCount < threshold && count >= threshold;

    return ok_(buildStampResult_(p.event, student.studentId, count, {
      company: company.companyName,
      justCleared,
    }));
  } finally {
    lock.releaseLock();
  }
}

// アクション: getStampProgress (stampToken)
// ============================================================

function actionGetStampProgress_(p) {
  if (!p.token && !p.pk) return err_('missing_token');
  if (!p.event) return err_('missing_event');

  if (!isEventDay_(p.event)) return err_('event_ended');

  // stampToken で学生を特定。未登録(STAMP_PARTICIPANTS未作成)の場合は cardToken(pk)で fallback。
  let student = p.token ? findStudentByStampToken_(p.token, p.event) : null;
  if (!student && p.pk) student = findStudentByCardToken_(p.pk, p.event);
  if (!student) return err_('invalid_token');

  const sheet = eventSs_(p.event).getSheetByName(SHEET.STAMP_LOG);
  const rows  = sheet.getLastRow() > 1
    ? sheet.getRange(2, 1, sheet.getLastRow() - 1, 9).getValues()
    : [];

  const myStamps = rows
    .filter(r => String(r[1]) === String(p.event) && String(r[2]) === String(student.studentId))
    .map(r => ({
      company:   r[6],
      companyId: String(r[7] || ''),   // スタンプ帳グリッドのID突合用
      time: r[0] instanceof Date
        ? Utilities.formatDate(r[0], Session.getScriptTimeZone(), 'yyyy/MM/dd HH:mm')
        : String(r[0]),
    }));

  return ok_(buildStampResult_(p.event, student.studentId, myStamps.length, {
    stamps: myStamps,
    companies: getCompaniesForEvent_(p.event),  // スタンプ帳の全ブース（ロゴ含む）
    name:      student.name,        // 本人確認用（共有端末での取り違え防止）
    studentId: student.studentId,   // 同上・補助識別
  }));
}

// イベントの全企業（スタンプ帳グリッド用・表示順）
// stampRally が明示的に 'FALSE' の企業はグリッドから除外する
function getCompaniesForEvent_(eventId) {
  const sheet = eventSs_(eventId).getSheetByName(SHEET.COMPANIES);
  if (!sheet || sheet.getLastRow() < 2) return [];
  const data    = sheet.getDataRange().getValues();
  const headers = data[0].map(String);
  const g       = (row, name) => { const i = headers.indexOf(name); return i >= 0 ? row[i] : ''; };
  const orderCol = headers.indexOf('表示順');
  const rows = data.slice(1)
    .filter(row => String(g(row, 'stampRally')).toUpperCase() !== 'FALSE')
    .map(row => ({
      companyId: String(g(row, 'companyId')),
      name:      String(g(row, '企業名')),
      logoUrl:   String(g(row, 'logoUrl') || ''),
      _order:    orderCol >= 0 ? Number(row[orderCol]) || 0 : 0,
    }));
  rows.sort((a, b) => a._order - b._order);
  return rows.map(({ _order, ...c }) => c);
}

// ============================================================
// アクション: exchangePrize (学生側確定 / stampToken)        ← T3-d 新規実装
// 学生が progress.html で景品交換を確定する。1人1回のみ。
// ============================================================

function actionExchangePrize_(p) {
  if (!p.token && !p.pk) return err_('missing_token');
  if (!p.event) return err_('missing_event');
  if (isEventInactive_(p.event)) return err_('event_inactive');
  const cfg = getConfig_(p.event);
  if (!isExchangeOpen_(cfg)) return err_('exchange_closed');
  // stampToken で学生特定。未登録なら cardToken(pk)で fallback（学生確定→studentIdで記録）
  let student = p.token ? findStudentByStampToken_(p.token, p.event) : null;
  if (!student && p.pk) student = findStudentByCardToken_(p.pk, p.event);
  if (!student) return err_('invalid_token');

  // 同時リクエストによる二重交換を防ぐためロック内で再計算してから書き込む
  const lock = LockService.getScriptLock();
  lock.waitLock(10000);
  try {
    const count  = countStamps_(p.event, student.studentId);
    const result = buildStampResult_(p.event, student.studentId, count, {});
    if (result.claimableNow <= 0) return err_('nothing_to_claim');
    eventSs_(p.event).getSheetByName(SHEET.PRIZE_LOG).appendRow([
      new Date(), p.event, student.studentId, student.name,
      count, '学生確定', result.claimableNow,
    ]);
    // appendRow後に countExchanged_ が更新されるので再計算して返す
    return ok_(buildStampResult_(p.event, student.studentId, count, {}));
  } finally {
    lock.releaseLock();
  }
}

  if (!p.event) return err_('missing_event');

  }

  const student = findStudentByCardToken_(p.token, p.event);
  if (!student) return err_('invalid_token');

  const count  = countStamps_(p.event, student.studentId);
  const result = buildStampResult_(p.event, student.studentId, count, {});

  return ok_({
    studentId:     student.studentId,
    name:          student.name,
    school:        student.school,
    category:      student.category,
    stampCount:    count,
    prizeUnitSize: result.prizeUnitSize,
    maxPrizes:     result.maxPrizes,
    eligibleTotal: result.eligibleTotal,
    exchangedCount:result.exchangedCount,
    claimableNow:  result.claimableNow,
    nextThreshold: result.nextThreshold,
    cleared:       result.cleared,
    exchanged:     result.exchanged,
  });
}

// ============================================================
// アクション: markPrizeExchanged (景品交換画面 / スタッフ)
// 学生を景品交換済みとして記録(1人1回のみ)
// ============================================================

function actionMarkPrizeExchanged_(p) {
  if (!p.key)   return err_('missing_key');
  if (!p.token) return err_('missing_token');
  if (!p.event) return err_('missing_event');

  const cfg = getConfig_(p.event);
  if (!cfg.exchangeKey || String(cfg.exchangeKey) !== String(p.key)) {
    return err_('invalid_staff_key');
  }

  const student = findStudentByCardToken_(p.token, p.event);
  if (!student) return err_('invalid_token');

  const lock = LockService.getScriptLock();
  try { lock.waitLock(6000); } catch (e) { return err_('lock_timeout'); }
  try {
    const count  = countStamps_(p.event, student.studentId);
    const result = buildStampResult_(p.event, student.studentId, count, {});

    if (result.claimableNow <= 0) return err_('nothing_to_claim');

    eventSs_(p.event).getSheetByName(SHEET.PRIZE_LOG).appendRow([
      new Date(), p.event, student.studentId, student.name,
      count, p.staff || '', result.claimableNow,
    ]);

    return ok_({
      studentId:     student.studentId,
      name:          student.name,
      stampCount:    count,
      claimedNow:    result.claimableNow,
      exchangedCount: result.exchangedCount + result.claimableNow,
      exchanged:     (result.exchangedCount + result.claimableNow) >= result.maxPrizes,
    });
  } finally {
    lock.releaseLock();
  }
}

// アクション: getSchoolList（大学マスターの名前一覧を返す）
// ============================================================

function actionGetSchoolList_(p) {
  // コード順に並べる。コード先頭2桁は読み頭文字のポケベル値（行→段）なので、
  // コード昇順 ≒ 五十音順（頭文字基準）。漢字名でも正しい五十音順で並ぶ。
  // ただし わ/を/ん は 01/05/00、仮コードは 0001〜 と先頭桁が 0 で、そのままだと
  // 先頭に来てしまう。先頭桁 '0' を '9' の後ろ（':'）に送るキーで末尾側へ補正する。
  const sortKey = c => {
    c = String(c).padStart(4, '0');
    return (c.charAt(0) === '0' ? ':' : c.charAt(0)) + c.substring(1);
  };
  const { uniMap } = loadUniversityMaster_();
  const schools = Array.from(uniMap.entries())
    .sort((a, b) => {
      const ka = sortKey(a[1]), kb = sortKey(b[1]);
      return ka < kb ? -1 : ka > kb ? 1 : 0; // localeCompareは':'を特殊扱いするためコードポイント比較
    })
    .map(e => e[0]);
  return ok_({ schools });
}

/**
 * 学生が景品交換した累計景品数
 * PRIZE_LOG の claimedCount 列を合計する。
 * 旧形式（claimedCount列なし）は行数をカウント（1行=1個扱い）。
 */
function countExchanged_(eventId, studentId) {
  const sheet = eventSs_(eventId).getSheetByName(SHEET.PRIZE_LOG);
  if (!sheet || sheet.getLastRow() < 2) return 0;

  const data    = sheet.getDataRange().getValues();
  const headers = data[0].map(String);
  const evCol   = headers.indexOf('eventId');
  const sidCol  = headers.indexOf('studentId');
  const cntCol  = headers.indexOf('claimedCount');

  return data.slice(1)
    .filter(r => String(r[evCol]) === String(eventId) && String(r[sidCol]) === String(studentId))
    .reduce((sum, r) => sum + (cntCol >= 0 ? Number(r[cntCol]) || 1 : 1), 0);
}

/** スタンプ数を数える */
function countStamps_(eventId, studentId) {
  const sheet = eventSs_(eventId).getSheetByName(SHEET.STAMP_LOG);
  const rows  = sheet.getLastRow() > 1
    ? sheet.getRange(2, 1, sheet.getLastRow() - 1, 9).getValues()
    : [];
  return rows.filter(r => String(r[1]) === String(eventId) && String(r[2]) === String(studentId)).length;
}

/**
 * スタンプ結果の共通レスポンスを組み立てる（累積交換モデル）
 *
 * 景品ロジック:
 *   eligibleTotal = min( floor(stampCount / prizeUnitSize), maxPrizes )
 *   claimableNow  = max( 0, eligibleTotal - exchangedCount )
 *   cleared       = claimableNow > 0
 *   exchanged     = exchangedCount >= maxPrizes（これ以上交換不可）
 */
function buildStampResult_(eventId, studentId, stampCount, extra) {
  const cfg           = getConfig_(eventId);
  const prizeUnitSize = Number(cfg.prizeUnitSize) || Number(cfg.prizeThreshold) || 5;
  const maxPrizes     = Number(cfg.maxPrizes)     || Number(cfg.prizeCount)     || 3;
  const exchangedCount  = countExchanged_(eventId, studentId);
  const eligibleTotal   = Math.min(Math.floor(stampCount / prizeUnitSize), maxPrizes);
  const claimableNow    = Math.max(0, eligibleTotal - exchangedCount);
  // 次の交換閾値（これ以上交換不可なら null）
  const nextTier        = exchangedCount + 1;
  const nextThreshold   = nextTier <= maxPrizes ? nextTier * prizeUnitSize : null;

  return Object.assign({
    stampCount,
    prizeUnitSize,
    maxPrizes,
    eligibleTotal,
    exchangedCount,
    claimableNow,
    nextThreshold,
    cleared:  claimableNow > 0,
    exchanged: exchangedCount >= maxPrizes,
  }, extra || {});
}

//
// SpreadsheetApp.openById は GAS で最も重い操作の一つ。1リクエストで eventSs_ が
// 4〜5回呼ばれ、毎回マスター＋イベントの2回 openById ＋ EVENT_LIST 全読みが走る。
// doGet/doPost ごとに新しいJSコンテキストになるため、モジュール変数でリクエスト単位に
// メモ化しても他リクエストと衝突しない（同一インスタンス再利用時も Spreadsheet は
// ライブ参照なので最新データを返し整合性は保たれる）。

var SS_CACHE_      = null;   // master Spreadsheet をメモ化
var EVENT_SS_CACHE_ = {};    // eventId → Spreadsheet をメモ化

function masterSs_() {
  if (SS_CACHE_) return SS_CACHE_;
  const id = PropertiesService.getScriptProperties()
    .getProperty('MASTER_SPREADSHEET_ID');
  if (!id) throw new Error('MASTER_SPREADSHEET_IDが未設定です。initializeMaster()を実行してください。');
  SS_CACHE_ = SpreadsheetApp.openById(id);
  return SS_CACHE_;
}

function eventSs_(eventId) {
  const key = String(eventId);
  if (EVENT_SS_CACHE_[key]) return EVENT_SS_CACHE_[key];
  const sheet   = masterSs_().getSheetByName(SHEET.EVENT_LIST);
  const data    = sheet.getDataRange().getValues();
  const headers = data[0].map(String);
  const idCol   = headers.indexOf('eventId');
  const ssIdCol = headers.indexOf('スプレッドシートID');
  const row     = data.slice(1).find(r => String(r[idCol]) === String(eventId));
  if (!row) throw new Error('イベントが見つかりません: ' + eventId);
  const ss = SpreadsheetApp.openById(row[ssIdCol]);
  EVENT_SS_CACHE_[key] = ss;
  return ss;
}

function getConfig_(eventId) {
  const cache = CacheService.getScriptCache();
  const key   = 'cfg_' + eventId;
  const hit   = cache.get(key);
  if (hit) return JSON.parse(hit);

  const data   = eventSs_(eventId).getSheetByName(SHEET.CONFIG).getDataRange().getValues();
  const config = {};
  data.forEach(([k, v]) => { if (k) config[String(k)] = v; });
  cache.put(key, JSON.stringify(config), 300);
  return config;
}

// 企業の来訪学生一覧などの閲覧期限。
// 既定は「イベント終了日+2ヶ月」を下限とし、管理者がそれより後の publicDeadline を
// 明示設定した場合のみ延長として採用する（短縮は不可＝最低2ヶ月は閲覧可）。
function isPastDeadline_(cfg) {
  var deadline = null;
  if (cfg.endDate) {
    var end = new Date(cfg.endDate);
    if (!isNaN(end)) {
      deadline = new Date(end);
      deadline.setMonth(deadline.getMonth() + 2);
      deadline.setHours(23, 59, 59);
    }
  }
  if (cfg.publicDeadline) {
    var pd = new Date(cfg.publicDeadline);
    if (!isNaN(pd) && (!deadline || pd > deadline)) deadline = pd; // 2ヶ月後より後なら延長
  }
  if (!deadline) return false; // 終了日も期限も不明なら無期限
  return new Date() > deadline;
}

// アクション: getCurrentEvent (T6)
// 今日の日付に合致するアクティブなイベントを返す
// ============================================================

function actionGetCurrentEvent_(p) {
  const sheet = masterSs_().getSheetByName(SHEET.EVENT_LIST);
  if (!sheet || sheet.getLastRow() < 2) return err_('no_active_event');

  const data      = sheet.getDataRange().getValues();
  const headers   = data[0].map(String);
  const idCol     = headers.indexOf('eventId');
  const nameCol   = headers.indexOf('イベント名');
  const startCol  = headers.indexOf('開始日');
  const endCol    = headers.indexOf('終了日');
  const statusCol = headers.indexOf('状態');

  const today = new Date();
  today.setHours(0, 0, 0, 0);

  const row = data.slice(1).find(r => {
    // 「公開停止」(旧データ互換で「完了」も)は開催日内でも公開対象から除外する
    if (statusCol >= 0) {
      const st = String(r[statusCol]);
      if (st === '公開停止' || st === '完了') return false;
    }
    const start = new Date(r[startCol]); start.setHours(0, 0, 0, 0);
    const end   = new Date(r[endCol]);   end.setHours(23, 59, 59, 999);
    return today >= start && today <= end;
  });

  if (!row) return err_('no_active_event');

  return ok_({
    eventId:   String(row[idCol]),
    eventName: row[nameCol],
    startDate: row[startCol] instanceof Date
      ? Utilities.formatDate(row[startCol], Session.getScriptTimeZone(), 'yyyy/MM/dd')
      : String(row[startCol]),
    endDate: row[endCol] instanceof Date
      ? Utilities.formatDate(row[endCol], Session.getScriptTimeZone(), 'yyyy/MM/dd')
      : String(row[endCol]),
  });
}

// ============================================================
// アクション: registerWalkIn (T6)
// 当日飛び込み参加者を登録してstampTokenを発行する。
// 同日・同一学生(氏名+大学名)の重複登録は既存stampTokenを返す。
// ============================================================

function actionRegisterWalkIn_(p) {
  if (!p.event) return err_('missing_event');
  if (!p.name || !p.school) return err_('missing_params');

  // サーバー側検証: イベント状態・開催日チェック
  if (isEventInactive_(p.event)) return err_('event_inactive');
  if (!isEventDay_(p.event))     return err_('not_event_day');

  const studentSheet = eventSs_(p.event).getSheetByName(SHEET.STUDENTS);

  // 重複チェック → 採番 → appendRow を同一ロック内で完結させ、同時登録での ID 重複を防ぐ
  const lock = LockService.getScriptLock();
  lock.waitLock(15000);
  let outcome_; // ロック内で確定した結果を外へ持ち出す
  try {
    const sData   = studentSheet.getDataRange().getValues();
    const sHeaders = sData[0].map(String);
    const nameCol  = sHeaders.indexOf('氏名');
    const schoolCol = sHeaders.indexOf('大学名');
    const sidCol   = sHeaders.indexOf('studentId');

    // 同日・同一学生の重複チェック(氏名+大学名)
    // 空白の表記揺れ（半角/全角/前後）で別人扱いにならないよう正規化して照合する。
    // （クッキー喪失→再登録での復帰の取りこぼしを防ぐ）
    const norm_ = s => String(s == null ? '' : s).replace(/[\s　]/g, '');
    const existing = sData.slice(1).find(r =>
      norm_(r[nameCol])   === norm_(p.name) &&
      norm_(r[schoolCol]) === norm_(p.school)
    );

    if (existing) {
      const existStudentId = String(existing[sidCol]);
      let participantSheet = eventSs_(p.event).getSheetByName(SHEET.STAMP_PARTICIPANTS);
      if (!participantSheet) {
        participantSheet = eventSs_(p.event).insertSheet(SHEET.STAMP_PARTICIPANTS);
        participantSheet.appendRow(['時刻', 'eventId', 'studentId', 'stampToken', '開始日時', 'status']);
      }
      const pData    = participantSheet.getDataRange().getValues();
      const pHeaders = pData[0].map(String);
      const pSidCol  = pHeaders.indexOf('studentId');
      const pStCol   = pHeaders.indexOf('stampToken');
      const pRow     = pData.slice(1).find(r =>
        String(r[pSidCol]) === existStudentId &&
        String(r[pHeaders.indexOf('eventId')]) === String(p.event)
      );
      // 既存学生だが参加者行が無い場合（例: 事前登録済みで未開始）は
      // 新規トークンを発行し、必ずシートへ追記する。
      // （追記しないと未保存トークンを返し、以後スタンプ不能になるため）
      let stampToken;
      if (pRow) {
        stampToken = String(pRow[pStCol]);
      } else {
        stampToken = Utilities.getUuid();
        participantSheet.appendRow([
          new Date(), p.event, existStudentId, stampToken, new Date(), 'active',
        ]);
      }
      const existCardToken = String(existing[sHeaders.indexOf('cardToken')] || '');
      const count = countStamps_(p.event, existStudentId);
      const result = buildStampResult_(p.event, existStudentId, count, { stampToken, cardToken: existCardToken });
      outcome_ = { type: 'existing', existStudentId, existCardToken, stampResult: result };
    } else {
      // 新規: studentId 採番・両シートへの appendRow をロック内で完結
      const yearMatch  = String(p.event).match(/(\d{4})/);
      const yearSuffix = yearMatch ? yearMatch[1].slice(-1) : String(new Date().getFullYear()).slice(-1);
      const { uniMap, maxAutoCode } = loadUniversityMaster_();
      const schoolName = String(p.school).trim();
      let uniCode = uniMap.get(schoolName);
      if (!uniCode) {
        uniCode = String(maxAutoCode + 1).padStart(4, '0');
        saveNewUniversities_([[schoolName, uniCode, '自動割当(要確認)']]);
      }
      // 連番は「西暦+大学コード+区分文字(=V)」で始まる既存IDの最大連番+1（＝同一大学・同一区分のみ）。
      // 全件通算ではなく大学・区分単位で 01 から採番。max+1 なので既存IDの歯抜けとも衝突しない。
      const idPrefix  = yearSuffix + uniCode + 'V';
      const studentId = idPrefix + String(nextStudentSeq_(sData, sidCol, idPrefix)).padStart(2, '0');
      const token     = Utilities.getUuid();

      // 学生マスターに追加
      const newRow = sHeaders.map(h => {
        const map = {
          'studentId': studentId, 'cardToken': token,  // T11: token → cardToken
          '氏名': String(p.name).trim(), 'ふりがな': String(p.furigana || '').trim(),
          '大学名': String(p.school).trim(), '学部学科': p.department || '',
          '学年': p.year || '', '属性': '一般参加学生',
          '自動車部在籍年数': p.clubYears || '',
          '住所(都道府県)': p.prefecture || '',
          '性別': p.gender || '',                       // フォーム準拠で追加
          '生年月日': p.birthday || '', 'メールアドレス': p.email || '',
          '電話番号': p.phone || '',
          '所属校出場': p.competing || '',              // 所属校が出場するか
          '規則誓約同意': p.rulesConsent  === 'true' ? 'true' : 'false', // 同意記録(大会規則書・誓約書)
          'SNS同意':     p.snsConsent     === 'true' ? 'true' : 'false', // 同意記録
          '個人情報同意': p.privacyConsent === 'true' ? 'true' : 'false', // 同意記録
          '登録種別': '当日',
        };
        return map[h] !== undefined ? map[h] : '';
      });
      studentSheet.appendRow(newRow);
      const phoneCol = sHeaders.indexOf('電話番号');
      if (phoneCol >= 0) {
        studentSheet.getRange(studentSheet.getLastRow(), phoneCol + 1).setNumberFormat('@');
      }

      // スタンプ参加者シートに追加（シート未存在時は自動作成）
      const stampToken = Utilities.getUuid();
      let stampParticipantSheet = eventSs_(p.event).getSheetByName(SHEET.STAMP_PARTICIPANTS);
      if (!stampParticipantSheet) {
        stampParticipantSheet = eventSs_(p.event).insertSheet(SHEET.STAMP_PARTICIPANTS);
        stampParticipantSheet.appendRow(['時刻', 'eventId', 'studentId', 'stampToken', '開始日時', 'status']);
      }
      stampParticipantSheet.appendRow([
        new Date(), p.event, studentId, stampToken, new Date(), 'active',
      ]);
      outcome_ = { type: 'new', studentId, token, stampToken };
    }
  } finally {
    lock.releaseLock();
  }

  // ロック解放後にメール送信（IO待ちでロックを保持しない）
  if (outcome_.type === 'existing') {
    const { existStudentId, existCardToken, stampResult } = outcome_;
    // 再登録は実質「復帰」操作。個人ページ(氏名+QR)のリンクを再送（失敗しても登録判定は維持）
    try {
      sendWalkInPassMail_(p.email, p.name, p.event, buildPassUrl_(p.appBase, existCardToken, p.event));
    } catch (err) {
      console.error('sendWalkInPassMail_ (already_registered) failed:', err);
    }
    return { ok: false, error: 'already_registered', data: stampResult,
             message: 'すでに本日のイベントに登録済みです' };
  } else {
    const { studentId, token, stampToken } = outcome_;
    // 個人ページ(氏名+QR)のリンクをメール送信（失敗しても登録は成立させる）
    try {
      sendWalkInPassMail_(p.email, p.name, p.event, buildPassUrl_(p.appBase, token, p.event));
    } catch (err) {
      console.error('sendWalkInPassMail_ failed:', err);
    }
    return ok_(buildStampResult_(p.event, studentId, 0, { stampToken, isNew: true, cardToken: token }));
  }
}

// ============================================================
// アクション: registerPreStudent (事前登録 / 脱Googleフォーム)
// 会期前に学生を事前登録する。公開（codeなし）。
// - STUDENTS に本人確認サブセットを upsert（登録種別=事前・cardToken発行）
// - 「事前登録」シートにフル記録（区分別項目・ファイルURLは後フェーズ）
// - 確認メールを送信（QR名刺は載せない）
// stampToken は発行しない（当日 activateStamp で発行）。
// ============================================================

function actionRegisterPreStudent_(p) {
  if (!p.event) return err_('missing_event');
  if (!p.name || !p.school || !p.email) return err_('missing_params');

  // サーバー側検証: イベント状態・受付期間・区分別締切
  if (isEventInactive_(p.event))  return err_('event_inactive');
  const cfg  = getConfig_(p.event);
  if (isBeforeFormOpen_(cfg))     return err_('form_not_open');
  // 参加区分を先行確定して締切チェックに使う（後段でも同じ値を流用）
  const attr = preAttr_(p.category, p.driverClass, p.serviceClass);
  if (isPastCategoryDeadline_(cfg, attr)) return err_('registration_closed');

  const ss           = eventSs_(p.event);
  const studentSheet = ss.getSheetByName(SHEET.STUDENTS);

  // 重複チェック → 採番 → STUDENTS appendRow を同一ロック内で完結させ、同時登録での ID 重複を防ぐ
  const lock = LockService.getScriptLock();
  lock.waitLock(15000);
  let studentId, cardToken;
  try {
    const sData   = studentSheet.getDataRange().getValues();
    const sHeaders = sData[0].map(String);
    const idx     = n => sHeaders.indexOf(n);

    // 重複（氏名+大学名+登録種別=事前）→ 既存 cardToken を返す
    const existing = sData.slice(1).find(r =>
      String(r[idx('氏名')])     === String(p.name).trim() &&
      String(r[idx('大学名')])   === String(p.school).trim() &&
      String(r[idx('登録種別')]) === '事前'
    );
    if (existing) {
      return { ok: false, error: 'already_registered',
        data: { studentId: String(existing[idx('studentId')]), cardToken: String(existing[idx('cardToken')]) },
        message: 'すでに事前登録済みです' };
    }

    // 参加区分 → クラスコード（attr は上で採番前に確定済み）
    const classCode = PRE_CLASS_CODES[attr] || 'V';

    // studentId 生成
    const yearMatch  = String(p.event).match(/(\d{4})/);
    const yearSuffix = yearMatch ? yearMatch[1].slice(-1) : String(new Date().getFullYear()).slice(-1);
    const { uniMap, maxAutoCode } = loadUniversityMaster_();
    const schoolName = String(p.school).trim();
    let uniCode = uniMap.get(schoolName);
    if (!uniCode) {
      uniCode = String(maxAutoCode + 1).padStart(4, '0');
      saveNewUniversities_([[schoolName, uniCode, '自動割当(要確認)']]);
    }
    // 連番は「西暦+大学コード+区分文字」で始まる既存IDの最大連番+1。
    // 属性ではなく studentId 接頭辞で数えるため、同一文字に集約される区分
    // （補欠/メカニック/応援＝S）どうしの衝突も防げる。max+1 で歯抜けとも衝突しない。
    const idPrefix = yearSuffix + uniCode + classCode;
    studentId      = idPrefix + String(nextStudentSeq_(sData, idx('studentId'), idPrefix)).padStart(2, '0');
    cardToken      = Utilities.getUuid();

    // STUDENTS upsert（本人確認サブセット）
    const newRow = sHeaders.map(h => {
      const map = {
        'studentId': studentId, 'cardToken': cardToken,
        '氏名': String(p.name).trim(), 'ふりがな': String(p.furigana || '').trim(),
        '大学名': String(p.school).trim(), '学部学科': p.department || '',
        '学年': p.year || '', '属性': attr, '自動車部在籍年数': p.clubYears || '',
        '住所(都道府県)': p.prefecture || '', '生年月日': p.birthday || '',
        '性別': p.gender || '', 'メールアドレス': p.email || '', '電話番号': p.phone || '',
        '所属校出場': p.competing || '',
        '規則誓約同意': p.rulesConsent   === 'true' ? 'true' : 'false',
        'SNS同意':     p.mediaConsent    === 'true' ? 'true' : 'false',
        '個人情報同意': p.privacyConsent  === 'true' ? 'true' : 'false',
        '登録種別': '事前',
      };
      return map[h] !== undefined ? map[h] : '';
    });
    studentSheet.appendRow(newRow);
    const phoneIdx = idx('電話番号');
    if (phoneIdx >= 0) {
      studentSheet.getRange(studentSheet.getLastRow(), phoneIdx + 1).setNumberFormat('@');
    }
  } finally {
    lock.releaseLock();
  }

  // ファイル（Drive保存・非公開）→ URLを p に設定
  // 権限未承認/一時障害でも登録自体は成立させる（サーバーエラーにしない）
  let fileUrls = {};
  let filesSaved = true;
  try {
    fileUrls = savePreRegFiles_(p.event, studentId, p.files, String(p.name || '').trim());
  } catch (err) {
    filesSaved = false;
    console.error('savePreRegFiles_ failed:', err);
  }
  p.pledgeUrl    = fileUrls.pledgeUrl    || '';
  p.licenseUrl   = fileUrls.licenseUrl   || '';
  p.insuranceUrl = fileUrls.insuranceUrl || '';
  p.approvalUrl  = fileUrls.approvalUrl  || '';

  // 事前登録シート（フル記録）
  appendPreRegRow_(ss, p, studentId, cardToken);

  // 確認メール（QR名刺は載せない・失敗しても登録は成立）
  let mailSent = true;
  try {
    sendPreRegMail_(p.email, p.name, p.event);
  } catch (err) {
    mailSent = false;
    console.error('sendPreRegMail_ failed:', err);
  }

  return ok_({ studentId, cardToken, registered: true, filesSaved, mailSent });
}

// studentId 採番ヘルパー: idPrefix（西暦+大学コード+区分文字）で始まる既存IDの
// 連番部の最大値+1 を返す。該当なしは 1（=01）。歯抜けがあっても衝突しない。
function nextStudentSeq_(sData, sidCol, idPrefix) {
  if (sidCol < 0) return 1;
  let maxSeq = 0;
  for (let i = 1; i < sData.length; i++) {
    const s = String(sData[i][sidCol] || '');
    if (s.startsWith(idPrefix)) {
      const n = Number(s.substring(idPrefix.length));
      if (!isNaN(n) && n > maxSeq) maxSeq = n;
    }
  }
  return maxSeq + 1;
}

// 参加区分(＋ドライバー登録区分) → 属性
function preAttr_(category, driverClass, serviceClass) {
  switch (String(category || '')) {
    case '出場選手(FGクラスドライバー)': return String(driverClass || 'Aドライバー');
    case '出場選手(女子クラスドライバー)': return '女子クラスドライバー';
    case '補欠ドライバー':                 return '補欠ドライバー';
    case '見学・応援学生(メカニック登録含む)':
      return String(serviceClass || '') === 'メカニック' ? 'メカニック' : '応援学生';
    default: return '応援学生';
  }
}

    '氏名': p.name || '', 'ふりがな': p.furigana || '', '性別': p.gender || '', '生年月日': p.birthday || '',
    'メールアドレス': p.email || '', '電話番号': p.phone || '', '郵便番号': p.postal || '',
    '都道府県': p.prefecture || '', '住所詳細': p.address || '',
    'ドライバー登録区分': p.driverClass || '', '自動車部在籍年数': p.clubYears || '',
    'シムアタック受験日': p.simDate || '', 'シムアタックタイム': p.simTime || '', '意気込み': p.comment || '',

  });

// 確認メール送信（GASオーナーGmailから。失敗しても登録は成立させる）
function sendPreRegMail_(to, name, eventId) {
  if (!to) return;
  try {
    const cfg    = getConfig_(eventId);
    const evName = cfg.eventName || eventId;
    // 件名・本文は設定シート(preRegMailSubject/preRegMailBody)で編集可。未設定なら既定文面。
    const subjTpl = cfg.preRegMailSubject ||
      '【{eventName}】学生参加 事前登録を受け付けました';
    const bodyTpl = cfg.preRegMailBody ||
      ('{name} 様\n\n' +
       '{eventName} の学生参加 事前登録を受け付けました。\n\n' +
       '当日は受付でご本人確認のうえ、入場パスをお渡しします。\n' +
       '登録内容に変更がある場合は事務局までご連絡ください。\n\n' +
       '── FORMULA GYMKHANA 事務局');
    const fill = s => String(s)
      .replace(/\{name\}/g, name || '')
      .replace(/\{eventName\}/g, evName);
    MailApp.sendEmail({ to: String(to), subject: fill(subjTpl), body: fill(bodyTpl) });
  } catch (err) {
    console.error('sendPreRegMail_ error:', err);
  }
}

// 当日参加者の個人ページ(氏名+QR)URLを組み立てる。
// appBase は静的サイトの app/ ディレクトリの絶対URL（フロントが location から渡す）。
// 取得できない場合は空文字（メール側でリンク無し扱い）。
function buildPassUrl_(appBase, token, eventId) {
  let base = String(appBase || '').trim();
  if (!base || !token) return '';
  if (base.charAt(base.length - 1) !== '/') base += '/';
  return base + 'mypass.html?token=' + encodeURIComponent(token) +
         '&event=' + encodeURIComponent(eventId);
}

// 当日参加者へ個人ページ(氏名+QR)のリンクをメール送信。
// QR画像は埋め込まず、リンクのみ（cardTokenを外部サービスに送らない方針）。
function sendWalkInPassMail_(to, name, eventId, passUrl) {
  if (!to) return;
  try {
    const cfg    = getConfig_(eventId);
    const evName = cfg.eventName || eventId;
    // 件名・本文は設定シート(walkInMailSubject/walkInMailBody)で編集可。未設定なら既定文面。
    // 差し込み: {name}＝氏名、{eventName}＝大会名、{passUrl}＝マイページURL。
    const subjTpl = cfg.walkInMailSubject ||
      '【{eventName}】当日参加 受付完了（あなたのMY PASS）';
    const bodyTpl = cfg.walkInMailBody ||
      ('{name} 様\n\n' +
       '{eventName} の当日参加登録を受け付けました。スタンプラリーにご参加いただけます。\n\n' +
       '▼あなたのMY PASS（氏名・QRコード）\n' +
       '{passUrl}\n\n' +
       '・このページをブックマーク／ホーム画面に追加しておくと、いつでも開けます。\n' +
       '・企業ブースの方には、このページのQRコードを見せてください。\n' +
       '・スタンプの進捗が消えてしまった場合も、このページから元に戻せます。\n\n' +
       '── FORMULA GYMKHANA 事務局');
    const safeUrl = passUrl || '（リンクの発行に失敗しました。会場スタッフにお問い合わせください）';
    const fill = s => String(s)
      .replace(/\{name\}/g, name || '')
      .replace(/\{eventName\}/g, evName)
      .replace(/\{passUrl\}/g, safeUrl);
    MailApp.sendEmail({ to: String(to), subject: fill(subjTpl), body: fill(bodyTpl) });
  } catch (err) {
    console.error('sendWalkInPassMail_ error:', err);
  }
}

// 管理画面: 当日参加者へ個人ページのリンクを再送信する。
function actionAdminResendWalkInMail_(p) {
  const e = validateAdmin_(p);
  if (e) return e;
  if (!p.event) return err_('missing_event');
  if (!p.token) return err_('missing_token');

  const student = findStudentByCardToken_(p.token, p.event);
  if (!student)        return err_('invalid_token');
  if (!student.email)  return err_('no_email');

  sendWalkInPassMail_(student.email, student.name, p.event,
                      buildPassUrl_(p.appBase, p.token, p.event));
  return ok_({ sent: true, email: student.email });
}

// 事前登録の提出ファイルをDriveに保存（非公開＝オーナー/組織のみ閲覧）。URLを返す。
// 階層: マイドライブ/FG事前登録書類/<eventId>/<書類種別>/  に保存（イベント・種別で混在防止）
function savePreRegFiles_(eventId, studentId, files, name) {
  const urls = {};
  if (!files || typeof files !== 'object') return urls;
  const eventFolder = getPreRegEventFolder_(eventId);
  const map = { pledge: '誓約書', license: '免許証', insurance: '保険証明', approval: '承諾書' };
  Object.keys(map).forEach(k => {
    const f = files[k];
    if (f && f.b64) {
      try {
        const sub   = getOrCreateChildFolder_(eventFolder, map[k]);
        const bytes = Utilities.base64Decode(f.b64);
        const fname = [studentId, name, (f.name || 'file')].filter(String).join('_');
        const blob  = Utilities.newBlob(bytes, f.mime || 'application/octet-stream', fname);
        const created = sub.createFile(blob); // 既定の権限＝非公開（個人情報のため共有しない）
        urls[k + 'Url'] = created.getUrl();
      } catch (err) { console.error('savePreRegFiles_', k, err); }
    }
  });
  return urls;
}

// 親「FG事前登録書類」→ イベント別サブフォルダ を取得（無ければ作成）
function getPreRegEventFolder_(eventId) {
  const root = getOrCreateChildFolder_(DriveApp.getRootFolder(), 'FG事前登録書類');
  return getOrCreateChildFolder_(root, String(eventId));
}

// 指定フォルダ直下に同名の子フォルダを取得（無ければ作成）
function getOrCreateChildFolder_(parent, name) {
  const it = parent.getFoldersByName(name);
  return it.hasNext() ? it.next() : parent.createFolder(name);
}

// ポケベルコード上2桁: 1桁目=行(あ=1…ら=9,わ=0)、2桁目=段(あいうえお=1〜5)。
var POKEBELL_TABLE_ = {
  'あ':'11','い':'12','う':'13','え':'14','お':'15',
  'か':'21','き':'22','く':'23','け':'24','こ':'25',
  'さ':'31','し':'32','す':'33','せ':'34','そ':'35',
  'た':'41','ち':'42','つ':'43','て':'44','と':'45',
  'な':'51','に':'52','ぬ':'53','ね':'54','の':'55',
  'は':'61','ひ':'62','ふ':'63','へ':'64','ほ':'65',
  'ま':'71','み':'72','む':'73','め':'74','も':'75',
  'や':'81','ゆ':'83','よ':'85',
  'ら':'91','り':'92','る':'93','れ':'94','ろ':'95',
  'わ':'01','を':'05','ん':'00'
};
// 濁音・半濁音・小書き → 清音ベースへ正規化（同志社=ど→と と同じ扱い）
var KANA_BASE_TABLE_ = (function () {
  var m = {'ぁ':'あ','ぃ':'い','ぅ':'う','ぇ':'え','ぉ':'お','っ':'つ','ゃ':'や','ゅ':'ゆ','ょ':'よ','ゎ':'わ'};
  'がぎぐげござじずぜぞだぢづでどばびぶべぼ'.split('').forEach(function (c) {
    m[c] = String.fromCharCode(c.charCodeAt(0) - 1);
  });
  'ぱぴぷぺぽ'.split('').forEach(function (c) {
    m[c] = String.fromCharCode(c.charCodeAt(0) - 2);
  });
  return m;
})();

function pokebellPrefix_(reading) {
  if (!reading) return '';
  var c = String(reading).trim().charAt(0);
  if (c >= 'ァ' && c <= 'ヶ') c = String.fromCharCode(c.charCodeAt(0) - 0x60); // カタカナ→ひらがな
  c = KANA_BASE_TABLE_[c] || c;
  return POKEBELL_TABLE_[c] || '';
}

// 読みの頭文字からポケベルコードを生成し、既存コードを見て連番採番して確定する
function actionAdminConfirmUniversity_(p) {
  var e = validateAdmin_(p);
  if (e) return e;
  var name    = String(p.name || '').trim();
  var prefix  = pokebellPrefix_(p.reading);
  if (!name) return err_('missing_params');
  if (!prefix) return err_('invalid_reading');

  var sheet = masterSs_().getSheetByName(SHEET.UNI_MASTER);
  if (!sheet || sheet.getLastRow() < 2) return err_('university_not_found');
  var data      = sheet.getDataRange().getValues();
  var headers   = data[0].map(String);
  var nameCol   = headers.indexOf('大学名');
  var codeCol   = headers.indexOf('コード');
  var statusCol = headers.indexOf('状態');

  // 同じ上2桁を持つ既存コードの最大連番 +1 で採番
  var maxSeq = 0;
  var targetRow = -1;
  var oldCode = '';
  var usedCodes = new Set();
  data.slice(1).forEach(function (row, i) {
    // 全角数字を半角に正規化してから比較
    var raw  = String(row[codeCol] || '').trim().replace(/[０-９]/g, function(c) { return String.fromCharCode(c.charCodeAt(0) - 0xFEE0); });
    var code = raw.padStart(4, '0');
    usedCodes.add(code);
    if (code.substring(0, 2) === prefix) {
      var seq = Number(code.substring(2));
      if (!isNaN(seq) && seq > maxSeq) maxSeq = seq;
    }
    if (String(row[nameCol] || '').trim() === name) {
      targetRow = i + 2;           // 1始まり+ヘッダ
      oldCode   = code;            // 書き換え対象の旧（仮）コード
    }
  });
  if (targetRow < 0) return err_('university_not_found');

  // 重複しないコードが見つかるまで連番を進める
  var nextSeq = maxSeq + 1;
  while (usedCodes.has(prefix + String(nextSeq).padStart(2, '0'))) { nextSeq++; }
  var newCode = prefix + String(nextSeq).padStart(2, '0');
  sheet.getRange(targetRow, codeCol + 1).setValue("'" + newCode); // 先頭0保持のため文字列化
  if (statusCol >= 0) sheet.getRange(targetRow, statusCol + 1).setValue('確定');

  // 仮コードで発行済みの学生IDを確定コードへ全シート横断で書き換える
  var rewritten = (oldCode && oldCode !== newCode) ? rewriteStudentUniCode_(oldCode, newCode) : 0;

  return ok_({ code: newCode, rewritten: rewritten });
}

// 大学コード確定時、旧（仮）コードで発行済みの学生IDを確定コードへ全シート横断で書き換える。
// studentId = 西暦(1) + 大学コード(4) + 区分(1) + 連番(2)。大学コード部のみ差し替える。
// studentId を持つ全シート（学生マスター/スタンプログ/参加者/景品/事前登録）を全イベント分走査。
// 確定コードは新規採番なので未使用 → 連番衝突は起きない。
function rewriteStudentUniCode_(oldCode, newCode) {
  oldCode = String(oldCode).padStart(4, '0');
  newCode = String(newCode).padStart(4, '0');
  if (oldCode === newCode) return 0;

  var SHEETS_WITH_SID = [SHEET.STUDENTS, SHEET.STAMP_LOG, SHEET.STAMP_PARTICIPANTS, SHEET.PRIZE_LOG, SHEET.PRE_REG];
  var evSheet = masterSs_().getSheetByName(SHEET.EVENT_LIST);
  if (!evSheet || evSheet.getLastRow() < 2) return 0;
  var evData    = evSheet.getDataRange().getValues();
  var evIdCol   = evData[0].map(String).indexOf('eventId');
  var updated   = 0;

  evData.slice(1).forEach(function (evRow) {
    var eventId = String(evRow[evIdCol] || '').trim();
    if (!eventId) return;
    var ss;
    try { ss = eventSs_(eventId); } catch (e) { return; }

    SHEETS_WITH_SID.forEach(function (sheetName) {
      var sheet = ss.getSheetByName(sheetName);
      if (!sheet || sheet.getLastRow() < 2) return;
      var data   = sheet.getDataRange().getValues();
      var sidCol = data[0].map(String).indexOf('studentId');
      if (sidCol < 0) return;
      for (var i = 1; i < data.length; i++) {
        var s = String(data[i][sidCol] || '');
        if (s.length >= 5 && s.substring(1, 5) === oldCode) {
          var nid = s[0] + newCode + s.substring(5);
          sheet.getRange(i + 1, sidCol + 1).setValue("'" + nid); // 文字列保持
          updated++;
        }
      }
    });
  });
  return updated;
}

// 学生情報を全シート横断で一括修正（氏名/ふりがな/大学名/属性/メールアドレス）
function actionAdminUpdateStudent_(p) {
  var e = validateAdmin_(p);
  if (e) return e;

  var sid = String(p.studentId || '').trim();
  if (!sid) return err_('missing_params');

  // 渡されたフィールドのみ更新
  var fields = {};
  if (p.name     !== undefined) fields.name     = String(p.name).trim();
  if (p.furigana !== undefined) fields.furigana  = String(p.furigana).trim();
  if (p.school   !== undefined) fields.school    = String(p.school).trim();
  if (p.category !== undefined) fields.category  = String(p.category).trim();
  if (p.email    !== undefined) fields.email     = String(p.email).trim();
  if (!Object.keys(fields).length) return err_('missing_params');

  // 各シートの列名マッピング
  var masterMap = { name:'氏名', furigana:'ふりがな', school:'大学名', category:'属性', email:'メールアドレス' };
  var preMap    = { name:'氏名', furigana:'ふりがな', school:'大学名', email:'メールアドレス' }; // category は事前登録には書かない

  var updated = { students: 0, preReg: 0, stampLog: 0, prizeLog: 0 };

  var evSheet  = masterSs_().getSheetByName(SHEET.EVENT_LIST);
  if (!evSheet || evSheet.getLastRow() < 2) return err_('no_events');
  var evData   = evSheet.getDataRange().getValues();
  var evIdCol  = evData[0].map(String).indexOf('eventId');
  var eventIds = evData.slice(1).map(function(r) { return String(r[evIdCol] || '').trim(); }).filter(Boolean);

  function updateSheet_(sheet, hdrs, data, sidCol, colMap) {
    for (var i = 1; i < data.length; i++) {
      if (String(data[i][sidCol]) !== sid) continue;
      Object.keys(colMap).forEach(function(key) {
        if (fields[key] === undefined) return;
        var c = hdrs.indexOf(colMap[key]);
        if (c >= 0) sheet.getRange(i + 1, c + 1).setValue(fields[key]);
      });
      return true;
    }
    return false;
  }

  eventIds.forEach(function(eventId) {
    var ss;
    try { ss = eventSs_(eventId); } catch (e2) { return; }

    // 学生マスター
    var stuSheet = ss.getSheetByName(SHEET.STUDENTS);
    if (stuSheet && stuSheet.getLastRow() > 1) {
      var d = stuSheet.getDataRange().getValues();
      var h = d[0].map(String);
      if (updateSheet_(stuSheet, h, d, h.indexOf('studentId'), masterMap)) updated.students++;
    }

    // 事前登録
    var preSheet = ss.getSheetByName(SHEET.PRE_REG);
    if (preSheet && preSheet.getLastRow() > 1) {
      var d = preSheet.getDataRange().getValues();
      var h = d[0].map(String);
      if (updateSheet_(preSheet, h, d, h.indexOf('studentId'), preMap)) updated.preReg++;
    }

    // スタンプログ (氏名/大学名/属性のみ)
    var stmpSheet = ss.getSheetByName(SHEET.STAMP_LOG);
    if (stmpSheet && stmpSheet.getLastRow() > 1) {
      var d = stmpSheet.getDataRange().getValues();
      var h = d[0].map(String);
      var sidC = h.indexOf('studentId');
      for (var i = 1; i < d.length; i++) {
        if (String(d[i][sidC]) !== sid) continue;
        if (fields.name)     { var c = h.indexOf('氏名');   if (c>=0) stmpSheet.getRange(i+1,c+1).setValue(fields.name); }
        if (fields.school)   { var c = h.indexOf('大学名'); if (c>=0) stmpSheet.getRange(i+1,c+1).setValue(fields.school); }
        if (fields.category) { var c = h.indexOf('属性');   if (c>=0) stmpSheet.getRange(i+1,c+1).setValue(fields.category); }
        updated.stampLog++;
      }
    }

    // 景品交換ログ (氏名のみ)
    var prizeSheet = ss.getSheetByName(SHEET.PRIZE_LOG);
    if (prizeSheet && prizeSheet.getLastRow() > 1) {
      var d = prizeSheet.getDataRange().getValues();
      var h = d[0].map(String);
      var sidC = h.indexOf('studentId');
      for (var i = 1; i < d.length; i++) {
        if (String(d[i][sidC]) !== sid) continue;
        if (fields.name) { var c = h.indexOf('氏名'); if (c>=0) prizeSheet.getRange(i+1,c+1).setValue(fields.name); }
        updated.prizeLog++;
      }
    }
  });

  return ok_({ updated: updated });
}

// イベントの基本情報（名前・開始日・終了日・状態）とCONFIG期間設定を更新
function actionAdminUpdateEvent_(p) {
  const e = validateAdmin_(p);
  if (e) return e;
  if (!p.eventId) return err_('missing_params');

  // EVENT_LIST を更新（getCurrentEvent の日付比較用）
  const sheet   = masterSs_().getSheetByName(SHEET.EVENT_LIST);
  const data    = sheet.getDataRange().getValues();

  const idCol   = headers.indexOf('eventId');

  const rowIdx = data.findIndex((r, i) => i > 0 && String(r[idCol]) === String(p.eventId));
  if (rowIdx < 0) return err_('event_not_found');

  const sheetRow = rowIdx + 1;
  const setList = (colName, value) => {
    const col = headers.indexOf(colName);
    if (col >= 0 && value) sheet.getRange(sheetRow, col + 1).setValue(value);
  };
  if (p.eventName) setList('イベント名', p.eventName);
  if (p.startDate) setList('開始日', p.startDate);
  if (p.endDate)   setList('終了日', p.endDate);

  if (p.stampStartAt || p.stampEndAt || p.exchangeDeadline) {
    const cfg = eventSs_(p.eventId).getSheetByName(SHEET.CONFIG);
    const cfgData = cfg.getDataRange().getValues();
    const setConfig = (key, value) => {
      if (!value) return;
      const idx = cfgData.findIndex(r => String(r[0]) === key);
      if (idx >= 0) cfg.getRange(idx + 1, 2).setValue(value);
      else cfg.appendRow([key, value]);
    };
    if (p.stampStartAt)     setConfig('stampStartAt',     p.stampStartAt);
    if (p.stampEndAt)       setConfig('stampEndAt',       p.stampEndAt);
    if (p.exchangeDeadline) setConfig('exchangeDeadline', p.exchangeDeadline);
  }

  CacheService.getScriptCache().remove('cfg_' + String(p.eventId));
  return ok_({ updated: true });
}

// 統計（学生数・スタンプ数・参加者数・景品交換数）

      eData.slice(1).forEach(r => {
        if (elsCol >= 0) coLunchSat += Number(r[elsCol]) || 0;
        if (ellCol >= 0) coLunchSun += Number(r[ellCol]) || 0;
      });

  }

  const stampSheet       = ss.getSheetByName(SHEET.STAMP_LOG);
  const participantSheet = ss.getSheetByName(SHEET.STAMP_PARTICIPANTS);

// 企業フィールドの更新（logoUrl / stampRally）列が無ければ末尾に自動作成
function actionAdminUpdateCompany_(p) {
  const e = validateAdmin_(p);
  if (e) return e;
  if (!p.event || !p.companyId) return err_('missing_params');

  const ALLOWED = ['logoUrl', 'stampRally'];
  const updates = {};
  ALLOWED.forEach(k => { if (p[k] !== undefined) updates[k] = p[k]; });
  if (!Object.keys(updates).length) return err_('missing_params');

  const sheet   = eventSs_(p.event).getSheetByName(SHEET.COMPANIES);
  const data    = sheet.getDataRange().getValues();
  let   headers = data[0].map(String);
  const idCol   = headers.indexOf('companyId');

  const rowIdx = data.slice(1).findIndex(r => String(r[idCol]) === String(p.companyId));
  if (rowIdx < 0) return err_('not_found');

  Object.entries(updates).forEach(([key, val]) => {
    let col = headers.indexOf(key);
    if (col < 0) {
      sheet.getRange(1, headers.length + 1).setValue(key);
      col = headers.length;
      headers.push(key);
    }
    sheet.getRange(rowIdx + 2, col + 1).setValue(String(val));
  });

  CacheService.getScriptCache().remove('cfg_' + p.event);
  return ok_({ companyId: p.companyId, ...updates });
}

// 企業ロゴを Google Drive にアップロードして thumbnail URL を返す
function actionAdminUploadCompanyLogo_(p) {
  const e = validateAdmin_(p);
  if (e) return e;
  if (!p.event || !p.companyId || !p.base64) return err_('missing_params');

  // Step1: Drive フォルダ確保
  let folder;
  try {
    const root = getOrCreateChildFolder_(DriveApp.getRootFolder(), 'FGロゴ');
    folder     = getOrCreateChildFolder_(root, String(p.event));
  } catch (er) {
    console.error('logo_step1_folder:', er);
    return { ok: false, error: 'logo_step1_folder_failed', message: 'Step1(Driveフォルダ): ' + (er && er.message ? er.message : String(er)) };
  }

  // Step2: 既存ファイル削除
  try {
    const existing = folder.getFilesByName(String(p.companyId));
    while (existing.hasNext()) existing.next().setTrashed(true);
  } catch (er) {
    console.error('logo_step2_trash:', er);
    // 削除失敗は致命的でないので続行
  }

  // Step3: ファイル作成
  let file;
  try {
    const mime  = p.mimeType || 'image/png';
    const ext   = mime.split('/')[1] || 'png';
    const bytes = Utilities.base64Decode(p.base64);
    const blob  = Utilities.newBlob(bytes, mime, String(p.companyId) + '.' + ext);
    file = folder.createFile(blob);
  } catch (er) {
    console.error('logo_step3_create:', er);
    return { ok: false, error: 'logo_step3_create_failed', message: 'Step3(ファイル作成): ' + (er && er.message ? er.message : String(er)) };
  }

  // Step4: 公開設定
  try {
    file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
  } catch (er) {
    console.error('logo_step4_sharing:', er);
    return { ok: false, error: 'logo_step4_sharing_failed', message: 'Step4(公開設定): ' + (er && er.message ? er.message : String(er)) };
  }

  const fileId = file.getId();
  const url    = 'https://drive.google.com/thumbnail?id=' + fileId + '&sz=s200';

  // Step5: シートの logoUrl 更新
  try {
    const sheet   = eventSs_(p.event).getSheetByName(SHEET.COMPANIES);
    const data    = sheet.getDataRange().getValues();
    let   headers = data[0].map(String);
    const idCol   = headers.indexOf('companyId');
    const rowIdx  = data.slice(1).findIndex(r => String(r[idCol]) === String(p.companyId));
    if (rowIdx >= 0) {
      let col = headers.indexOf('logoUrl');
      if (col < 0) {
        sheet.getRange(1, headers.length + 1).setValue('logoUrl');
        col = headers.length;
      }
      sheet.getRange(rowIdx + 2, col + 1).setValue(url);
      CacheService.getScriptCache().remove('cfg_' + p.event);
    }
  } catch (er) {
    console.error('logo_step5_sheet:', er);
    return { ok: false, error: 'logo_step5_sheet_failed', message: 'Step5(シート更新): ' + (er && er.message ? er.message : String(er)) };
  }

  return ok_({ url, fileId });
}

// ============================================================
// 管理者パスワード復旧（メール宛リセット）
// ロックアウト時用の公開フロー。認証は「事前設定した復旧用メール受信」で担保。
//   1) adminRequestReset … 復旧用メールへ再設定コードを送信
//   2) adminConfirmReset  … コード＋新パスワードで ADMIN_KEY を再設定
// 最終フォールバックとして resetAdminKey()（エディタ実行）も用意。
// ============================================================

var ADMIN_RESET_TTL_MS_    = 15 * 60 * 1000;  // コード有効期限: 15分
var ADMIN_RESET_RESEND_MS_ = 60 * 1000;       // 再送制限: 60秒
var ADMIN_RESET_MAX_TRIES_ = 5;               // 検証失敗の上限

// 暗号学的乱数(UUID)由来の8文字コード（紛らわしい文字を避けるため16進大文字）
function genResetCode_() {
  return Utilities.getUuid().replace(/-/g, '').slice(0, 8).toUpperCase();
}

function sha256Hex_(s) {
  var bytes = Utilities.computeDigest(
    Utilities.DigestAlgorithm.SHA_256, String(s), Utilities.Charset.UTF_8);
  return bytes.map(function (b) { return ('0' + (b & 0xff).toString(16)).slice(-2); }).join('');
}

function clearAdminReset_() {
  var props = PropertiesService.getScriptProperties();
  ['ADMIN_RESET_HASH', 'ADMIN_RESET_EXPIRES', 'ADMIN_RESET_ATTEMPTS', 'ADMIN_RESET_SENT_AT']
    .forEach(function (k) { props.deleteProperty(k); });
}

// 公開: 復旧用メール宛に再設定コードを送信
function actionAdminRequestReset_(p) {
  var props = PropertiesService.getScriptProperties();
  var email = props.getProperty('ADMIN_RECOVERY_EMAIL');
  // 復旧用メール未設定でも、設定有無を外部に漏らさないため一律 ok を返す
  if (!email) return ok_({ requested: true });
  // 再送レート制限（連続送信によるメール爆撃を防ぐ）
  var lastSent = Number(props.getProperty('ADMIN_RESET_SENT_AT') || 0);
  if (Date.now() - lastSent < ADMIN_RESET_RESEND_MS_) return ok_({ requested: true });

  var code = genResetCode_();
  props.setProperty('ADMIN_RESET_HASH', sha256Hex_(code));   // 平文は保存しない
  props.setProperty('ADMIN_RESET_EXPIRES', String(Date.now() + ADMIN_RESET_TTL_MS_));
  props.setProperty('ADMIN_RESET_ATTEMPTS', '0');
  props.setProperty('ADMIN_RESET_SENT_AT', String(Date.now()));

  try {
    MailApp.sendEmail({
      to: String(email),
      subject: '【FG管理パネル】パスワード再設定コード',
      body: [
        '管理パネルのパスワード再設定がリクエストされました。',
        '',
        '再設定コード: ' + code,
        '（有効期限: 15分・1回限り有効）',
        '',
        '管理パネルのログイン画面で、このコードと新しいパスワードを入力してください。',
        'お心当たりがない場合は、このメールを破棄してください。',
        '',
        '── FORMULA GYMKHANA 事務局システム',
      ].join('\n'),
    });
  } catch (err) {
    console.error('actionAdminRequestReset_ mail error:', err);
  }
  return ok_({ requested: true });
}

// 公開: コード＋新パスワードで ADMIN_KEY を再設定
function actionAdminConfirmReset_(p) {
  if (!p.code || !p.newKey) return err_('missing_params');
  var props = PropertiesService.getScriptProperties();
  var hash  = props.getProperty('ADMIN_RESET_HASH');
  var exp   = Number(props.getProperty('ADMIN_RESET_EXPIRES') || 0);
  var tries = Number(props.getProperty('ADMIN_RESET_ATTEMPTS') || 0);

  if (!hash || !exp)            return err_('reset_not_requested');
  if (Date.now() > exp)         { clearAdminReset_(); return err_('reset_expired'); }
  if (tries >= ADMIN_RESET_MAX_TRIES_) { clearAdminReset_(); return err_('reset_locked'); }

  if (sha256Hex_(String(p.code).trim().toUpperCase()) !== hash) {
    props.setProperty('ADMIN_RESET_ATTEMPTS', String(tries + 1));
    return err_('reset_invalid_code');
  }

  props.setProperty('ADMIN_KEY', String(p.newKey).trim());
  clearAdminReset_();
  return ok_({ reset: true });
}

// ── GASエディタから直接実行する保守用関数（最終フォールバック） ──
// Apps Scriptエディタで関数を選んで実行する。GASを所有するGoogleアカウントでのみ
// 実行可能なため、パネルから締め出されても確実に復旧できる。
// 使い方: 関数 resetAdminKey を選び実行 → 実行ログに新しいキーが出る。
//        引数付きで呼ぶには一時的に resetAdminKey('好きなキー') と書いて実行。
function resetAdminKey(newKey) {
  var key = (newKey && String(newKey).trim()) || (genResetCode_() + genResetCode_());
  PropertiesService.getScriptProperties().setProperty('ADMIN_KEY', String(key));
  clearAdminReset_();
  Logger.log('ADMIN_KEY を再設定しました。新しいキー: ' + key);
  return key;
}

// 前回イベントから企業を引き継ぎ（stampKey/viewKey/logoUrl を保持）
function actionAdminImportCompanies_(p) {
  const e = validateAdmin_(p);
  if (e) return e;
  if (!p.event || !p.sourceEvent) return err_('missing_params');
  if (String(p.event) === String(p.sourceEvent)) return err_('same_event');

  const sourceSheet = eventSs_(p.sourceEvent).getSheetByName(SHEET.COMPANIES);
  if (!sourceSheet || sourceSheet.getLastRow() < 2) return err_('no_source_companies');

  const targetSheet  = eventSs_(p.event).getSheetByName(SHEET.COMPANIES);
  const sourceData   = sourceSheet.getDataRange().getValues();
  const sourceHdrs   = sourceData[0].map(String);
  const targetData   = targetSheet.getDataRange().getValues();
  const targetHdrs   = targetData[0].map(String);
  const targetIdCol  = targetHdrs.indexOf('companyId');
  const existingIds  = new Set(targetData.slice(1).map(r => String(r[targetIdCol])));

  const sg = (row, name) => { const i = sourceHdrs.indexOf(name); return i >= 0 ? row[i] : ''; };

  let imported = 0;
  const baseOrder = targetData.length; // ヘッダー込み行数 = 既存企業数+1
  sourceData.slice(1).forEach(row => {
    const companyId = String(sg(row, 'companyId'));
    if (!companyId || existingIds.has(companyId)) return;
    const newRow = targetHdrs.map(h => {
      switch (h) {
        case 'companyId':  return companyId;
        case '企業名':     return String(sg(row, '企業名'));
        case 'stampKey':   return String(sg(row, 'stampKey')  || '');
        case 'viewKey':    return String(sg(row, 'viewKey')   || '');
        case '表示順':     return baseOrder + imported;
        case 'hasNfc':     return String(sg(row, 'hasNfc')    || 'false');
        case 'logoUrl':    return String(sg(row, 'logoUrl')   || '');
        case 'stampRally': return String(sg(row, 'stampRally')|| 'true');
        default:           return '';
      }
    });
    targetSheet.appendRow(newRow);
    existingIds.add(companyId);
    imported++;
  });

  return ok_({ imported });
}

  } else {
    const evSheet = masterSs_().getSheetByName(SHEET.EVENT_LIST);
    if (!evSheet || evSheet.getLastRow() < 2) return err_('no_active_event');

    const today = new Date(); today.setHours(0,0,0,0);

      if (statCol >= 0 && (r[statCol] === '公開停止' || r[statCol] === '完了')) return false;
      const s = new Date(r[startCol]); s.setHours(0,0,0,0);
      const e = new Date(r[endCol]);   e.setHours(23,59,59,999);
      return today >= s && today <= e;
    });
    if (!evRow) return err_('no_active_event');
    eventId = String(evRow[idCol]);
  }

  const ss    = eventSs_(eventId);
  let   sheet = ss.getSheetByName(SHEET.COMPANY_ENTRIES);
  if (!sheet) {
    sheet = ss.insertSheet(SHEET.COMPANY_ENTRIES);
    sheet.appendRow([
      '申込日時','eventId',
      '企業名正式','社名略称','代表者名','担当者名','電話番号','担当者電話','メールアドレス',
      '郵便番号','都道府県','住所',
      '出展内容','ブース区画','展示車両数','デモ走行','デモ走行詳細',
      '人パス','車両パス','昼食土','昼食日',
      '備考','状態',
    ]);
    sheet.setFrozenRows(1);
  }

  const toInt = v => { const n = parseInt(v, 10); return isNaN(n) ? 0 : n; };
  const rowData = [
    new Date(), eventId,
    p.companyName, p.companyShort || '', p.director, p.contact, p.tel, p.contactTel, p.email,
    p.zip, p.prefecture || '', p.address,
    p.content, p.booth, toInt(p.carCount), p.demo, p.demoDetail || '',
    toInt(p.personPass), toInt(p.carPass), toInt(p.lunchSat), toInt(p.lunchSun),
    p.note || '', '受付済',
  ];
  const newRow = sheet.getLastRow() + 1;
  sheet.getRange(newRow, 1, 1, rowData.length).setValues([rowData]);
  // 電話番号(7列目), 担当者電話(8列目), 郵便番号(10列目) をテキスト書式に
  [7, 8, 10].forEach(c => sheet.getRange(newRow, c).setNumberFormat('@'));

  return ok_({ message: '受付しました' });
}

// ============================================================
// アクション: registerSchoolEntry（出場校エントリー提出）
// 重複時（同一学校名）は既存行を上書き、承諾書は履歴として蓄積。
// ============================================================

function actionRegisterSchoolEntry_(p) {
  if (!p.event) return err_('missing_event');
  if (isEventInactive_(p.event)) return err_('event_inactive');

  const cfg   = getConfig_(p.event);
  const state = schoolEntryFormState_(cfg);
  if (state === 'not_open') return err_('form_not_open');
  if (state === 'expired')  return err_('registration_closed');

  const required = ['schoolName','repName','repPhone','repEmail','shipName','shipPostal','shipAddress','shipPhone','carPassCount','schoolPermission','ruleConsent'];
  for (const k of required) {
    if (!p[k] && p[k] !== 0) return err_('missing_' + k);
  }

  const ss = eventSs_(p.event);
  let sheet = ss.getSheetByName(SHEET.SCHOOL_ENTRIES);
  if (!sheet) {
    sheet = ss.insertSheet(SHEET.SCHOOL_ENTRIES);
    sheet.getRange(1, 1, 1, SCHOOL_ENTRY_COLS.length).setValues([SCHOOL_ENTRY_COLS]);
    sheet.setFrozenRows(1);
  }

  const lock = LockService.getScriptLock();
  try { lock.waitLock(10000); } catch (e) { return err_('lock_timeout'); }
  try {
    const data    = sheet.getDataRange().getValues();
    const headers = data[0].map(String);
    const idx     = n => headers.indexOf(n);
    const nameCol = idx('学校名');

    // 承諾書ファイルを保存（あれば）
    let approvalUrl = '';
    if (p.approval && p.approval.b64) {
      try {
        const folder = getSchoolEntryEventFolder_(p.event);
        const bytes  = Utilities.base64Decode(p.approval.b64);
        const fname  = [Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyyMMdd-HHmmss'),
                        String(p.schoolName).trim(), (p.approval.name || 'approval')].join('_');
        const blob   = Utilities.newBlob(bytes, p.approval.mime || 'application/octet-stream', fname);
        approvalUrl  = folder.createFile(blob).getUrl();
      } catch (err) { console.error('school entry file save', err); }
    }

    const now = new Date();
    const normSchool = String(p.schoolName).trim();
    const existingRowIdx = data.slice(1).findIndex(r => String(r[nameCol]).trim() === normSchool);
    let isUpdate = false;
    let updateCount = 1;
    let historyStr = '';
    if (existingRowIdx >= 0) {
      isUpdate = true;
      const existRow = data[existingRowIdx + 1];
      const prevUrl  = String(existRow[idx('承諾書URL')] || '');
      const prevHist = String(existRow[idx('承諾書履歴')] || '');
      updateCount    = (Number(existRow[idx('更新回数')]) || 1) + 1;
      if (prevUrl) {
        const stamp = Utilities.formatDate(new Date(existRow[idx('提出日時')] || now), Session.getScriptTimeZone(), 'yyyy/MM/dd HH:mm');
        historyStr = (prevHist ? prevHist + '\n' : '') + `[${stamp}] ${prevUrl}`;
      } else {
        historyStr = prevHist;
      }
    }

    const finalApprovalUrl = approvalUrl || (isUpdate ? String(data[existingRowIdx + 1][idx('承諾書URL')] || '') : '');

    const rowMap = {
      '提出日時':       now,
      'eventId':        p.event,
      '学校名':          normSchool,
      '代表者氏名':      String(p.repName || '').trim(),
      '代表者電話':      String(p.repPhone || ''),
      '代表者メール':    String(p.repEmail || '').trim(),
      '発送先_名義':     String(p.shipName || '').trim(),
      '発送先_郵便番号': String(p.shipPostal || ''),
      '発送先_住所':     String(p.shipAddress || '').trim(),
      '発送先_電話':     String(p.shipPhone || ''),
      '車両入場証枚数':  String(p.carPassCount || ''),
      '学校許可取得':    String(p.schoolPermission || ''),
      '事務局への連絡事項': String(p.note || ''),
      '承諾書URL':       finalApprovalUrl,
      '承諾書履歴':      historyStr,
      '規則同意':        p.ruleConsent === 'true' || p.ruleConsent === true ? 'true' : 'false',
      '更新回数':        updateCount,
    };
    const newRow = SCHOOL_ENTRY_COLS.map(h => rowMap[h] !== undefined ? rowMap[h] : '');

    let targetRow;
    if (isUpdate) {
      targetRow = existingRowIdx + 2; // header + 1-index
      sheet.getRange(targetRow, 1, 1, newRow.length).setValues([newRow]);
    } else {
      targetRow = sheet.getLastRow() + 1;
      sheet.getRange(targetRow, 1, 1, newRow.length).setValues([newRow]);
    }
    // 電話・郵便番号をテキスト書式に
    ['代表者電話','発送先_郵便番号','発送先_電話'].forEach(col => {
      const ci = SCHOOL_ENTRY_COLS.indexOf(col);
      if (ci >= 0) sheet.getRange(targetRow, ci + 1).setNumberFormat('@');
    });

    // 確認メール送信（非同期・失敗しても登録は成立）
    sendSchoolEntryConfirmMail_(p.event, {
      to:       p.repEmail,
      repName:  p.repName,
      school:   normSchool,
      isUpdate: isUpdate,
    });

    return ok_({ message: isUpdate ? '内容を更新しました' : '受付しました', isUpdate: isUpdate });
  } finally {
    lock.releaseLock();
  }
}

/** 出場校エントリー確認メール */
function sendSchoolEntryConfirmMail_(eventId, info) {
  if (!info.to) return;
  try {
    const cfg    = getConfig_(eventId);
    const evName = cfg.eventName || eventId;
    const subject = info.isUpdate
      ? `【${evName}】出場校エントリーの更新を受け付けました`
      : `【${evName}】出場校エントリーを受け付けました`;
    const body =
      `${info.repName || ''} 様\n\n` +
      `${evName} の出場校エントリー（${info.school}）を${info.isUpdate ? '更新' : '受付け'}ました。\n\n` +
      (info.isUpdate
        ? '以前の入力内容は破棄され、今回の内容が最新として保存されました。\n'
        : '内容に変更がある場合は、同じフォームから再度送信してください（最新のもので上書きされます）。\n') +
      `\n入場パスは後日発送いたします。\n\n── FORMULA GYMKHANA 事務局`;
    MailApp.sendEmail({ to: String(info.to), subject: subject, body: body });
  } catch (err) {
    console.error('sendSchoolEntryConfirmMail_ error:', err);
  }
}

// ============================================================
// アクション: adminUpdateEntry（出展申込の部分更新）
// ============================================================
function actionAdminUpdateEntry_(p) {
  const e = validateAdmin_(p);
  if (e) return e;
  if (!p.event)   return err_('missing_event');
  if (!p.company) return err_('missing_params');
  if (!p.updates || typeof p.updates !== 'object') return err_('missing_params');

  const sheet = eventSs_(p.event).getSheetByName(SHEET.COMPANY_ENTRIES);
  if (!sheet || sheet.getLastRow() < 2) return err_('not_found');

  const data    = sheet.getDataRange().getValues();
  const headers = data[0].map(String);

  var rowIdx = -1;
  var nameCol = headers.indexOf('企業名正式');
  var abbCol  = headers.indexOf('社名略称');
  for (var r = 1; r < data.length; r++) {
    if (String(data[r][nameCol]) === p.company || String(data[r][abbCol]) === p.company) {
      rowIdx = r;
      break;
    }
  }
  if (rowIdx < 0) return err_('not_found');

  var allowed = ['展示車両数','人パス','車両パス','昼食土','昼食日','デモ走行'];
  var changes = [];
  for (var key in p.updates) {
    if (allowed.indexOf(key) < 0) continue;
    var col = headers.indexOf(key);
    if (col < 0) continue;
    var oldVal = data[rowIdx][col];
    var newVal = p.updates[key];
    if (String(oldVal) !== String(newVal)) {
      changes.push(key + ': ' + oldVal + ' → ' + newVal);
      sheet.getRange(rowIdx + 1, col + 1).setValue(newVal);
    }
  }

  if (changes.length) {
    var logCol = headers.indexOf('変更履歴');
    if (logCol < 0) {
      logCol = headers.length;
      sheet.getRange(1, logCol + 1).setValue('変更履歴');
    }
    var tz = Session.getScriptTimeZone();
    var ts = Utilities.formatDate(new Date(), tz, 'MM/dd HH:mm');
    var prev = String(sheet.getRange(rowIdx + 1, logCol + 1).getValue() || '');
    var entry = '[' + ts + '] ' + changes.join(', ');
    sheet.getRange(rowIdx + 1, logCol + 1).setValue(prev ? prev + '\n' + entry : entry);
  }
  return ok_({ updated: true });
}
